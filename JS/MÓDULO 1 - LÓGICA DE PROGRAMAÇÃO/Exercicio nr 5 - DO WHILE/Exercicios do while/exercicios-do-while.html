1. Escreva um programa que solicita ao usuário que digite um número. O programa deve continuar solicitando um número até que o usuário digite um número par.

let number; do { number = prompt("Digite um número qualquer:"); } while (number % 2 == 1);

2. Escreva um programa que solicita ao usuário que adivinhe um número entre 1 e 100. O programa deve informar se o número digitado é maior ou menor que o número a ser adivinhado. O programa deve continuar pedindo ao usuário para adivinhar até que ele acerte o número.

let guess; let magicNumber = Math.floor(Math.random() * 100) + 1; do { guess = Number(prompt("Adivinhe o número entre 1 e 100: ")); if (guess < magicNumber) { alert("O número a ser adivinhado é maior do que o informado."); } else if (guess > magicNumber) { alert("O número a ser adivinhado é menor do que o informado."); } else { alert("Parabéns, você adivinhou o número"); break; } } while (guess !== magicNumber);

3. Escreva um programa que solicita ao usuário que digite uma senha. O programa deve continuar solicitando uma senha até que o usuário digite a senha correta.

let password = 12345; let digiteSenha; do { digiteSenha = Number(prompt("Digite a sua senha: ")); if (password === digiteSenha) { alert("Senha correta! Acesso liberado."); } else { alert("Senha incorreta. Tente novamente!"); } } while (digiteSenha !== password);

4. Escreva um programa que solicita ao usuário que digite uma palavra. O programa deve continuar solicitando uma palavra até que o usuário digite uma palavra que comece com a letra "A".

let digitePalavra; do { digitePalavra = prompt("Digite uma palavra: "); } while (digitePalavra[0] !== "A");
